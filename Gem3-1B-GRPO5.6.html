# Final GRPO System Prompt
SYSTEM_PROMPT = """Respond in the following format:
<reasoning>
...
</reasoning>
<answer>
...
</answer>

Return a positive numeric answer, choose a number from 1 to 3 (i.e. 1,2,3)."""

# Utility: extract '####' answer
def extract_hash_answer(text: str) -> Optional[str]:
    if "####" not in text:
        return None
    return text.split("####")[-1].strip()

# Restrict to valid single-digit answers
def is_single_digit_numeric(answer: Optional[str]) -> bool:
    return bool(answer and re.fullmatch(r"[1-3]", answer.strip())) # Restricted to 1-3 - nursery GSM8K

# Format dataset entry with reasoning and numeric response instructions
def format_grpo_example(example):
    try:
        prompt = f"{SYSTEM_PROMPT}\n\n{example['question']}"
        answer = extract_hash_answer(example["answer"])
        return {"prompt": prompt, "answer": answer}
    except Exception:
        return {"prompt": "", "answer": ""}

def apply_chat_format(example):
    chat = [
        {"role": "system", "content": "Answer the following question with a single number from 1 to 3 (i.e. 1,2,3)."},
        {"role": "user", "content": example["prompt"]},
        {"role": "assistant", "content": ""}
    ]
    formatted_prompt = tokenizer.apply_chat_template(chat, tokenize=False)
    return {
        "prompt": formatted_prompt,
        "answer": example["answer"]
    }

# Load GSM8K, format, filter, and apply chat template - Gemma badly needs it!
def get_grpo_dataset(split="train") -> Dataset:
    raw = load_dataset("openai/gsm8k", "main", split=split)
    formatted = raw.map(format_grpo_example)
    filtered = formatted.filter(lambda x: is_single_digit_numeric(x["answer"]))
    chat_formatted = filtered.map(apply_chat_format)
    return chat_formatted

from datasets import Dataset
import random

def split_dataset_in_half(dataset: Dataset, seed: int = 137):
    random.seed(seed)
    total_size = len(dataset)
    indices = list(range(total_size))
    random.shuffle(indices)

    half = total_size // 2
    indices1 = indices[:half]
    indices2 = indices[half:]

    subset1 = dataset.select(indices1)
    subset2 = dataset.select(indices2)

    return subset1, subset2

import re
import torch
from typing import List, Union, Dict, Any


def extract_final_answer(text: str) -> Union[str, None]:
    """
    Robustly extract a final numeric answer from model output.

    Args:
        text (str): The text output from the model.

    Returns:
        str or None: The extracted number as a string, or None if not found.
    """
    # Try extracting from tags like <final> or <answer>
    match = re.search(r"<(?:solution|final|answer)[^>]*>\s*(-?\d+)", text, re.IGNORECASE)
    if match:
        return match.group(1).strip()

    # Fallback: try extracting from the last 100 characters
    tail = text.strip()[-100:]
    match = re.search(r"(-?\d+)(?!\S)", tail)
    if match:
        return match.group(1).strip()

    return None

def hybrid_reward_func(
    prompts: List[str],
    completions: List[List[Dict[str, Any]]],
    answer: List[Union[str, int]],
    normalize: bool = True,
    **kwargs
) -> List[float]:
    """
    Compute the average reward for the completions of each prompt in the batch.
    Returns a list where each element is the average reward for a prompt,
    repeated num_generations times.
    """
    all_average_rewards = []
    num_generations = kwargs.get("num_generations", 1)

    #print("hybrid_reward_func started")

    for i, (prompt, completions_per_prompt, correct_answer) in enumerate(zip(prompts, completions, answer)):
        #print("=" * 50)
        #print(f"QUESTION {i + 1}:\n{prompt}")
        #print(f"TRUE ANSWER: {correct_answer}")
        #print(f"COMPLETIONS ({len(completions_per_prompt)}):\n{completions_per_prompt}")
        print(f"COMPLETIONS: {len(completions_per_prompt)}")

        correct = int(correct_answer)
        rewards = []

        for j, comp in enumerate(completions_per_prompt):
            if isinstance(comp, dict) and "content" in comp:
                response = comp["content"]
            else:
                #print(f"WARNING: Unexpected format at [{i}][{j}]: {comp}")
                response = str(comp)

            extracted = extract_final_answer(response)

            if extracted is None:
                rewards.append(-1.0)
            else:
                try:
                    pred = float(extracted)  # float to catch 0.0, 0.01, etc.
                    if abs(pred) < 0.1:  # Penalize zero-ish values
                        rewards.append(-1.0)
                    elif int(pred) == correct:
                        rewards.append(3.0)
                    elif abs(int(pred) - correct) == 1:
                        rewards.append(2.0)
                    elif 0 < pred <= 9:
                        rewards.append(1.0)
                    else:
                        rewards.append(-0.2)
                except ValueError:
                    rewards.append(-1.0)

        if rewards:
            average_reward = sum(rewards) / len(rewards)
            all_average_rewards.extend([average_reward] * num_generations)
            #print(f"Average reward for prompt {i + 1}: {average_reward}")
        else:
            all_average_rewards.extend([0.0] * num_generations)
            #print(f"No valid completions for prompt {i + 1}, average reward: 0.0")
        
        X_max = 3.0  # Maximum possible reward (adapt as needed)
        X_min = -1.0  # Minimum possible reward (adapt as needed)
        if normalize and len(all_average_rewards) > 1:
            r = torch.tensor(all_average_rewards)
            #mean = r.mean()
            #std = r.std() + 1e-6  # avoid divide by zero
            #r = (r - mean) / std
            r = 2 * (r - X_min) / (X_max - X_min) - 1
            all_average_rewards = r.tolist()

    print("Final all_average_rewards:", all_average_rewards)
    return all_average_rewards
