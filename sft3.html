def format_prompt(examples):
    """Format and tokenize a batch of prompts using Qwen2.5 chat template"""

    # Format chat using template
    prompts = [template_tokenizer.apply_chat_template(chat, tokenize=False) for chat in examples["messages"]]

    # Tokenize all prompts
    tokens = template_tokenizer(prompts, padding="max_length", truncation=True, max_length=512)

    # Add labels for training (for causal LM, labels = input_ids)
    tokens["labels"] = tokens["input_ids"].copy()

    return tokens
